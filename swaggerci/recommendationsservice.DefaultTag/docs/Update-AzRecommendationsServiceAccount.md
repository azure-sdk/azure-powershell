## SYNOPSIS

 [-Cor <ICorsRule[]>] [-EndpointAuthentication <IEndpointAuthentication[]>]
 [-IdentityType <ManagedServiceIdentityType>] [-IdentityUserAssignedIdentity <Hashtable>]
 [-ReportsConnectionString <String>] [-Tag <Hashtable>] [-DefaultProfile <PSObject>] [-AsJob] [-NoWait]
 [-Confirm] [-WhatIf] [<CommonParameters>]
