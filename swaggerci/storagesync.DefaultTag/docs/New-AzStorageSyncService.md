## SYNOPSIS

 [-SubscriptionId <String>] [-IdentityType <ManagedServiceIdentityType>]
 [-IdentityUserAssignedIdentity <Hashtable>] [-IncomingTrafficPolicy <IncomingTrafficPolicy>]
 [-Tag <Hashtable>] [-UseIdentity] [-DefaultProfile <PSObject>] [-AsJob] [-NoWait] [-Confirm] [-WhatIf]
 [<CommonParameters>]
